Here is a C++ program that merges two ordered arrays into a single ordered array:

```cpp
#include <iostream>
using namespace std;

// Function to merge two ordered arrays into a single ordered array
void mergeOrderedArrays(int array1[], int array2[], int m, int n, int result[]) {
    int i = 0; // Index for array1
    int j = 0; // Index for array2
    int k = 0; // Index for result array

    // Merge smaller elements first
    while (i < m && j < n) {
        if (array1[i] < array2[j]) {
            result[k] = array1[i];
            i++;
        } else {
            result[k] = array2[j];
            j++;
        }
        k++;
    }

    // If there are remaining elements in either array, append them to the result
    while (i < m) {
        result[k] = array1[i];
        i++;
        k++;
    }

    while (j < n) {
        result[k] = array2[j];
        j++;
        k++;
    }
}

// Function to print an array
void printArray(int array[], int size) {
    for (int i = 0; i < size; i++) {
        cout << array[i] << " ";
    }
    cout << endl;
}

int main() {
    int array1[] = {1, 3, 5, 7};
    int array2[] = {2, 4, 6, 8};

    int m = sizeof(array1) / sizeof(array1[0]);
    int n = sizeof(array2) / sizeof(array2[0]);

    // Create a result array with enough space to hold all elements
    int result[m + n];

    cout << "Array 1: ";
    printArray(array1, m);

    cout << "Array 2: ";
    printArray(array2, n);

    mergeOrderedArrays(array1, array2, m, n, result);

    cout << "Merged Array: ";
    printArray(result, m + n);

    return 0;
}
```
